# Python dependencies for the AG Coder agent (scripts/agent/run_agent.py).
#
# Install into the Python environment used to run the agent subprocess:
#   pip install -r scripts/agent/requirements.txt
#
# The API server spawns `python3 scripts/agent/run_agent.py`, so these must be
# importable from that interpreter's path.

# Event-driven graph runtime the agent is built on.
#
# The [postgres] extra pulls in psycopg 3.x, which backs ActiveGraph's NATIVE
# PostgresEventStore. The runtime persists its own authoritative event log to
# that store (see `_open_native_store` in run_agent.py); the UI's projection
# tables are derived from it rather than hand-written as a parallel trace.
activegraph[postgres]==1.0.5.post2

# psycopg 3.x — required by ActiveGraph's PostgresEventStore (the framework's
# durable, forkable, replayable event log). Normally installed by the
# [postgres] extra above; pinned here so it's explicit.
psycopg[binary]>=3.1,<4

# psycopg 2.x — the application's own connection to the UI projection tables
# (runs / agent_events / graph_objects / graph_relations). Distinct from the
# framework store's psycopg 3 connection; the two never share a connection.
psycopg2-binary>=2.9,<3

# Optional: enables ActiveGraph's PrometheusMetrics backend (the runtime feeds
# it counters/histograms/gauges). Without it the agent falls back to NoOpMetrics
# and the runtime.status() snapshot (ops.status event) still works. Safe to omit.
prometheus_client>=0.20

# LLM SDKs — used by ActiveGraph's AnthropicProvider / OpenAIProvider, which
# drive the @llm_behavior turn loop. The agent no longer calls these SDKs
# directly. Only needed for live model mode; without an API key the agent runs
# in deterministic demo mode and these are unused (but harmless to install).
anthropic>=0.105,<1
openai>=2,<3
